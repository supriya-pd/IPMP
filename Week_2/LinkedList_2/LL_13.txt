#include <bits/stdc++.h>
using namespace std;

struct Node {
    int data;
    struct Node *next;
    Node(int x) {
        data = x;
        next = NULL;
    }
};

Node *rotate(struct Node *head, int k);

void printList(Node *n)
{
    while (n != NULL)
    {
        cout<< n->data << " ";
        n = n->next;
    }
    cout<< endl;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n, val, k;
        cin>>n;
        
        cin>> val;
        Node *head = new Node(val);
        Node *tail = head;
        
        for(int i=0; i<n-1; i++)
        {
            cin>> val;
            tail->next = new Node(val);
            tail = tail->next;
        }
        
        cin>> k;
        
        head = rotate(head,k);
        printList(head);
    }
    return 1;
}







Node* rotate(Node* head, int k)
{
 Node *ptr=head;
 if(!ptr)
 return head;
 int c=1;
 while(ptr!=NULL && ptr->next!=NULL)
 {
     ptr=ptr->next;
 }
 ptr->next=head;
 ptr=head;
 while(c<k)
 {
     c++;
     ptr=ptr->next;
 }
 head=ptr->next;
 ptr->next=NULL;
 return head;
}
